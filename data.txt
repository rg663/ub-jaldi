    callback: function (set, suggestions) {
      if (suggestions.length) {
        jQuery('.tt-menu').addClass(set.collection.toLowerCase());
      } else {
        jQuery('.tt-menu').removeClass(set.collection.toLowerCase());
      }
      return suggestions;
    },
    datasets: {
      organic: {
        name: 'Suggestions',
        show: (window.innerWidth <= 720) ? '6' : '12',
        profile: '_default',
        collection: 'meta-search'
      },
      people: {
        name: 'People',
        show: (window.innerWidth <= 720) ? '3' : '4',
        profile: 'autocompletion',
        collection: 'db-people'
        , template: {
          suggestion: jQuery('#auto-completion-people').text()
        }
      },
      places: {
        name: 'Locations',
        show: (window.innerWidth <= 720) ? '3' : '8',
        profile: 'autocompletion',
        collection: 'db-places'
        , template: {
          suggestion: jQuery('#auto-completion-places').text()
        }
      }
    }
  });
  (function() {
    var topQueriesDiv = jQuery('#top-queries');

    if (topQueriesDiv[0]) {
      jQuery.ajax({
        url: '/admin-api/analytics/v1/collections/meta-search/queries/top',
        type: "get",
        data: {
          profile: '_default',
          earliestDate: '2022-02-06',
          latestDate: '2022-02-20'
        },
        success: processTopQueriesResponse,
        error: console.error
      });
    }

    function processTopQueriesResponse(response) {
      if (response && response.data && response.data.queries.length > 0) {
        var topQueriesListFirst = topQueriesDiv.find('.top-queries-list-first');
        var topQueriesListLast = topQueriesDiv.find('.top-queries-list-last');
        // Filter out any empty string queries (not likely to occur in produciton, but just a precaution).
        response.data.queries = response.data.queries.filter(function(item) {
          return item.query && item.query.trim().length !== 0;
        });
        for (var i = 0; i < response.data.queries.length; i++) {
          var item = response.data.queries[i];
          var topQueriesList = (i < 5 ? topQueriesListFirst : topQueriesListLast);
          topQueriesList.append('<li><a href="https://www.buffalo.edu/search/search.html?collection=meta-search&profile=_default&query=' + item.query + '">' + item.query + '</a></li>');
        }
        topQueriesDiv[0].classList.remove('no-top-queries');
      }
    }
  })();
  });
